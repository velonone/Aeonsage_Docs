  <Tab title="CLI">
    
    <div className="install-section">
      <h4>Install via Package Manager</h4>
      
      <CodeTabs>
        <CodeTab label="pnpm (Recommended)">
          ```bash
          pnpm add -g aeonsage
          ```
        </CodeTab>
        
        <CodeTab label="npm">
          ```bash
          npm install -g aeonsage
          ```
        </CodeTab>
        
        <CodeTab label="yarn">
          ```bash
          yarn global add aeonsage
          ```
        </CodeTab>
      </CodeTabs>
      
      <h4>Initialize and Start</h4>
      
      ```bash
      # Initialize configuration
      aeonsage init
      
      # Start the Gateway
      aeonsage gateway start
      
      # Gateway is now running
      ```
      
      <Output>
        ```
        [INFO] AeonSage Gateway v2026.2.15
        [INFO] Listening on http://127.0.0.1:GATEWAY_PORT
        [INFO] Admin token generated
        [INFO] Dashboard: http://localhost:GATEWAY_PORT
        ```
      </Output>
      
      <div className="next-step">
        <p>Open your browser to the dashboard URL shown in the output.</p>
      </div>
    </div>
    
  </Tab>
  
  <Tab title="Docker">
    
    <div className="install-section">
      <h4>Run with Docker</h4>
      
      ```bash
      docker run -d \
        --name aeonsage-gateway \
        -p GATEWAY_PORT:GATEWAY_PORT \
        -v aeonsage-data:/root/.aeonsage \
        -e GATEWAY_TOKEN=your-secure-token \
        -e OPENAI_API_KEY=your-key \
        aeonsage/gateway:latest
      ```
      
      <h4>Or use Docker Compose</h4>
      
      Create <code>docker-compose.yml</code>:
      
      ```yaml
      version: '3.8'
      services:
        aeonsage:
          image: aeonsage/gateway:latest
          ports:
            - "GATEWAY_PORT:GATEWAY_PORT"
          environment:
            - GATEWAY_TOKEN=${GATEWAY_TOKEN}
            - OPENAI_API_KEY=${OPENAI_API_KEY}
          volumes:
            - ./config:/root/.aeonsage
      ```
      
      ```bash
      docker-compose up -d
      ```
      
      <VerifyBadge status="success">
        Gateway running at http://localhost:GATEWAY_PORT
      </VerifyBadge>
    </div>
    
  </Tab>
</Tabs>
